use crate::{FaktoryJob, JobFailedError, ServerResponse};
use tokio::sync::oneshot;

/// Pipelined client command responses are sent here.
pub(crate) type CommandResponder = oneshot::Sender<ServerResponse>;

// We receive pipelined client command responses here.
// pub(crate) type CommandReceiver = oneshot::Receiver<ServerResponse>;

/// Picket Line commands. This is somewhat higher level than [crate::FaktoryCommand].
/// Essentially, here we are interested with commands from the _Picket Line_ API, as opposed to Faktory API.
#[derive(Debug)]
pub enum Command {
    /// A set of queues to fetch from, or the `default` queue.
    /// Queue priority is implicit via the vec ordering.
    /// TODO: This should be an [&str] probably?
    Fetch((CommandResponder, Option<Vec<String>>)),

    /// `Ack` a given job id.
    Ack((CommandResponder, String)),

    /// `Push` a given job
    /// TODO: Not sure if should be boxed or not tbqh...
    Push((CommandResponder, Box<FaktoryJob>)),

    /// Send a BEAT message
    Beat(CommandResponder),

    // `Fail` a given job
    Fail((CommandResponder, String, JobFailedError)),
}
